390 void BlobMainWindowView::scaleMainPanelImages( double scaleFactor )
391 {
392  if( scaleFactor < 1.0 ) {
393  scaleFactor = 1.0;
394  }
395  else if( scaleFactor > MAX_IMAGE_SCALE_FACTOR ) {
396  scaleFactor = MAX_IMAGE_SCALE_FACTOR;
397  }
398  int imageWidth = (int)(MIN_MAIN_IMAGE_WIDTH * scaleFactor),
399  imageHeight = (int)(MIN_MAIN_IMAGE_HEIGHT * scaleFactor),
400  widthIncrease = imageWidth - MIN_MAIN_IMAGE_WIDTH,
401  heightIncrease = imageHeight - MIN_MAIN_IMAGE_HEIGHT,
402  mainPanelWidth = mainPanelMinWidth_ + (2 * widthIncrease),
403  mainPanelHeight = mainPanelMinHeight_ + heightIncrease;
404 
405  debugPrintWidthHeight( "source image", imageWidth, imageHeight );
406  debugPrintWidthHeight( "tracked image", imageWidth, imageHeight );
407  debugPrintWidthHeight( "main panel", mainPanelWidth, mainPanelHeight );
408 
409  sourceGLWidget_->setFixedSize( imageWidth, imageHeight );
410  trackedGLWidget_->setFixedSize( imageWidth, imageHeight );
411  mainUi_->widget_mainPanel->setFixedSize( mainPanelWidth, mainPanelHeight );
412 }

475  frameH = screenH;
476  needToResize = true;
477  }

480  blobMainWindow_->resize( frameW, frameH );

505  if( frameY < screenY ) { frameY = screenY; }

507  debugPrintGeo( "Available", screenX, screenY, screenW, screenH );
508  debugPrintGeo( "Frame", frameX, frameY, frameW, frameH );
509  blobMainWindow_->move( frameX, frameY );
