139 void BlobMainWindow::frame( double deltaFrameTime )
140 {
141  sourceGLWidget_->frame( deltaFrameTime );
142 
143  if( needToCalibrate_ ) {
144  calibrationGLWidget_->frame( deltaFrameTime );
145  }
146  else if( sourceGLWidget_->hasNewFrame() ) {
147  trackedGLWidget_->frame( deltaFrameTime );
148  backgroundGLWidget_->frame( deltaFrameTime );

153  if( sourceGLWidget_->needToDisplayUpdatedFps() ) {
154  //debug_.printLn( sourceGLWidget_->getUpdatedFps().toStdString() );
155  ui_.textEdit_outputMessage->setText( sourceGLWidget_->getUpdatedFps() );
156  }
