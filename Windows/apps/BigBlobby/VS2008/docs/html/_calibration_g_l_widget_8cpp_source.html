128  case Qt::Key_W: coreVision->keyPressed( KeyEnum::KEY_w ); break;
129  case Qt::Key_S: coreVision->keyPressed( KeyEnum::KEY_s ); break;
130 
131  case Qt::Key_Right: coreVision->keyPressed( KeyEnum::KEY_ARROW_RIGHT );
132  if( event->isAutoRepeat() ) { QApplication::processEvents(); }
133  break;
134  case Qt::Key_Left: coreVision->keyPressed( KeyEnum::KEY_ARROW_LEFT );
135  if( event->isAutoRepeat() ) { QApplication::processEvents(); }
136  break;

139  break;

141  if( event->isAutoRepeat() ) { QApplication::processEvents(); }
142  break;
143  case Qt::Key_Equal: coreVision->keyPressed( KeyEnum::KEY_EQUAL_SIGN ); break;
144  case Qt::Key_Minus: coreVision->keyPressed( KeyEnum::KEY_MINUS ); break;
145  case Qt::Key_Plus: coreVision->keyPressed( KeyEnum::KEY_PLUS ); break;
146  case Qt::Key_Underscore: coreVision->keyPressed( KeyEnum::KEY_UNDERSCORE ); break;
147
