140 void CoreVision::update()
141 {
142  bNewFrame = false;
143  if( bExited ) { return; }
144 
145  if( bUseCamera ) {
146  if( bStartVideoCamera ) {
147  stopVideoPlayer();
148  clearVideoFrame();
149  initVideoCamera();
150  setUpImagesAndFilters();
151  learnBackground();
152  bStartVideoCamera = false;
153  }
154  #ifdef TARGET_WIN32
155  if( ps3Camera != NULL ) { //ps3 camera
156  bNewFrame = ps3Camera->isFrameNew();
157  }
158  else if( ffmvCamera != NULL ) {
159  ffmvCamera->grabFrame();
160  bNewFrame = true;
161  }
162  else if(videoGrabber != NULL) {
163  videoGrabber->grabFrame();
164  bNewFrame = videoGrabber->isFrameNew();
165  }
166  //else if( dsvlCamera != NULL ) {
167  // bNewFrame = dsvlCamera->isFrameNew();
168  //}
169  #else
170  videoGrabber->grabFrame();
171  bNewFrame = videoGrabber->isFrameNew();
172  #endif
173  }
174  else { // Video player requested or in use.
175  if( bStartVideoPlayer ) {
176  clearVideoFrame();
177  initVideoPlayer();
178  learnBackground();
179  bStartVideoPlayer = false;
180  }
181  videoPlayer->idleMovie();
182  bNewFrame = videoPlayer->isFrameNew();
183  }
184  processVideoFrame();
185 }

203 void CoreVision::drawFilter( FilterEnum::Enum filterType )
204 {
205  if( bExited ) { return; }
206 
207  glMatrixMode( GL_MODELVIEW );
208  glPushMatrix();
209  glTranslatef( 0.0f, videoHeight, 0.0f );
210  glRotatef( 180.0f, 1.0f, 0.0f, 0.0f );
211 
212  switch( filterType ) {
213  case FilterEnum::SOURCE: drawSourceImage(); break;
214  case FilterEnum::TRACKED: drawTrackedImage(); break;
215  case FilterEnum::BACKGROUND: filters.drawBackgoundImage(); break;
216  case FilterEnum::SMOOTH: filters.drawSmoothImage(); break;
217  case FilterEnum::HIGHPASS: filters.drawHighpassImage(); break;
218  case FilterEnum::AMPLIFY: filters.drawAmplifyImage(); break;
219  }
220  glPopMatrix();
221 }

1227 /*******************************************************************************
1228 This is the most important helper function for update(). If a new frame is
1229 available, this function will call on either the CPU or GPU frame grabber
1230 function, and then process the image by apply filters and finding any
1231 contours before calling on the TUIO object to send out tracking info.
1232 Number of frames and time for calculating fps (frames per second) are
1233 also tracked here.
1234 *******************************************************************************/
1235 void CoreVision::processVideoFrame()
1236 {
1237  if( bNewFrame ) {
1238  ++frames;
1239  int time = ofGetElapsedTimeMillis();
1240 
1241  if( time > (lastFpsLog + 1000) ) {
1242  fps = frames;
1243  frames = 0;
1244  lastFpsLog = time;
1245  bNeedToDisplayUpdatedFps = true;
1246  }
1247  float beforeTime = ofGetElapsedTimeMillis();
1248  int minArea = (minBlobSize * 2) + 1,
1249  maxArea = (int)((videoWidth * videoHeight * 0.4f) * (maxBlobSize * 0.001f) );
1250 
1251  if( bUseGpuShaders ) {
1252  grabFrameToGPU( filters.gpuSourceTex );
1253  filters.applyGPUFilters();
1254  contourFinder.findContours( filters.gpuReadBackImageGS,
1255  minArea, maxArea, maxNumberBlobs, false);
1256  }
1257  else { // CPU mode requested
1258  grabFrameToCPU();
1259  filters.applyCPUFilters( processedImage );
1260  contourFinder.findContours( processedImage,
1261  minArea, maxArea, maxNumberBlobs, false );
1262  }
1263  blobTracker->track( &contourFinder );
1264  differenceTime = ofGetElapsedTimeMillis() - beforeTime;
1265 
1266  if( filters.useAutoBackground() ) {
1267  filters.setBackgroundLearnRate( backgroundLearnRate * 0.001f );
1268 
1269  if( contourFinder.nBlobs > 0 ) { // Learn slow if blobs present.
1270  filters.setBackgroundLearnRate( backgroundLearnRate * 0.0001f );
1271  }
1272  }
1273  if( bTuioMode ) {
1274  tuio.sendTUIO( &(blobTracker->getTrackedBlobs()) );
1275  }
