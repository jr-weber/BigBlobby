105 void BlobXmlSettings::loadDefaultSettings()
106 {
107  blobIO::BlobParamsValidator * validator = NULL;
108 
109  try {
110  validator = blobParamsXmlReader_->read( DEFAULT_CONFIG_FILE.toStdString() );
111  }
112  catch( FileNotFoundException e ) {
113  dialogBoxUtils_->warnUser( e );
114  }
115  catch( BlobXmlReaderException e ) {
116  dialogBoxUtils_->warnUser( e );
117  }
118  catch( UnknownXmlTagException e ) {
119  dialogBoxUtils_->warnUser( e );
120  }
121  catch( BlobParamsException e ) {
122  dialogBoxUtils_->warnUser( e );
123  }
124  catch( ... ) {
125  QString msg = "An unexpected error occurred while trying to read \n'"
126  + DEFAULT_CONFIG_FILE + "'.\n"
127  + "Factory default settings will be used instead.\n";
128  dialogBoxUtils_->warnUser( msg );
129  }
130  if( validator == NULL ) {
131  validator = new blobIO::BlobParamsValidator();
132  QString msg = "\"Factory\" default settings will be used instead of "
133  "settings from the file '" + DEFAULT_CONFIG_FILE + "'.\n\n"
134  "If you call \"Quit (save settings)\" from the File menu "
135  "or click the close box, the factory default settings "
136  "will overwrite the DefaultBlobParams.xml, whereas "
137  "\"Exit (without saving)\" will leave the "
138  "DefaultBlobParams.xml file unchanged.";
139  dialogBoxUtils_->warnUser( msg );
140  }
141  quietlyUpdateGui( validator );
142  sourceGLWidget_->updateCoreVisionParams( validator );
143  debugPrint( validator );
144  delete validator;
145 }

255 {

259  sourceGLWidget_->getCoreVisionParams( &validator );
