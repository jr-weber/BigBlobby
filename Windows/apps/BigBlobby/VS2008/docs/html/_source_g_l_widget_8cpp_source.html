576 void SourceGLWidget::resizeGL( int viewportWidth, int viewportHeight )
577 {
578  int videoWidth = coreVision_.getVideoWidth(),
579  videoHeight = coreVision_.getVideoHeight();
580  float videoAspect = (float)videoWidth / videoHeight,
581  viewportAspect = (float)viewportWidth / viewportHeight;
582 
583  //debugPrintResizeGL( viewportWidth, viewportHeight, videoWidth, videoHeight );
584  glViewport( 0, 0, viewportWidth, viewportHeight );
585  glMatrixMode( GL_PROJECTION );
586  glLoadIdentity();
587 
588  if( viewportAspect > videoAspect ) {
589  // Viewport is too wide, so need some left/right padding (on x-axis)
590  int biggerWidth = (int)((viewportAspect / videoAspect) * videoWidth),
591  adjustment = (biggerWidth - videoWidth) / 2;
592  glOrtho( -adjustment, videoWidth + adjustment, 0, videoHeight, -1.0, 1.0 );
593  }
594  else if( videoAspect > viewportAspect ) {
595  // Viewport is too tall, so need some top/bottom padding (on y-axis).
596  int biggerHeight = (int)((videoAspect / viewportAspect) * videoHeight),
597  adjustment = (biggerHeight - videoHeight) / 2;
598  glOrtho( 0, videoWidth, -adjustment, videoHeight + adjustment, -1.0, 1.0 );
599  }
600  else { // Aspect ratios are equal.
601  glOrtho( 0, videoWidth, 0, videoHeight, -1.0, 1.0 );
602  }
603  glMatrixMode( GL_MODELVIEW );
604 }

674 
675  // Blob slots
676  setImageThreshold( paramsValidator_.getImageThreshold() );
677  setMinBlobSize( paramsValidator_.getMinBlobSize() );
678  setMaxBlobSize( paramsValidator_.getMaxBlobSize() );
679  setMinMovementThreshold( paramsValidator_.getMinMovementThreshold() );
680  setMaxNumberBlobs( paramsValidator_.getMaxNumberBlobs() );
681  useDarkBlobs( paramsValidator_.useDarkBlobs() );
682  showPressure( paramsValidator_.isShowPressureOn() );
683  showOutlines( paramsValidator_.isShowOutlinesOn() );
684  showLabels( paramsValidator_.isShowLabelsOn() );
685 
686  // Background and Dynamic Background slots
687  learnBackground();
688  useAutoBackground( paramsValidator_.useAutoBackground() );
689  setBackgroundLearnRate( paramsValidator_.getBackgroundLearnRate() );
690 
691  // Filters slots
692  useSmoothing( paramsValidator_.useSmoothing() );
693  setSmoothing( paramsValidator_.getSmoothing() );
694  useHighpass( paramsValidator_.useHighpass() );
695  setHighpassBlur( paramsValidator_.getHighpassBlur() );
696  setHighpassNoise( paramsValidator_.getHighpassNoise() );

698  setAmplify( paramsValidator_.getAmplify() );
699  useGpuShaders( paramsValidator_.useGpuShaders() );
700 
701  // Network slots
702  if( paramsValidator_.useNoOutput() ) {
703  useNoOutput( true );
704  }
705  else if( paramsValidator_.useTuioUdp() ) {
706  useTuioUdp( true );
707  }
708  else if( paramsValidator_.useFlashXml() ) {
709  useFlashXml( true );
710  }
711  else if( paramsValidator_.useBinaryTcp() ) {
712  useBinaryTcp( true );
713  }
714  else { // All 4 can not be set to false.
715  useNoOutput( true );
716  }
717  setupTuio( paramsValidator_.getLocalHost(),
718  paramsValidator_.getTuioPortOut(),
719  paramsValidator_.getTuioFlashPortOut() );
720 
721  coreVision_.setUpBlobCalibration();
722 }
