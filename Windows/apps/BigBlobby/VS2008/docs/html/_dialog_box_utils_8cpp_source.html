140 /*******************************************************************************
141 This private helper function expects a title that will go at the top of the
142 dialog box and a message that is already formatted.
143 
144 The message can include newline characters. A warning icon will be displayed
145 next to the message, and the dialog box is modal (blocks all user input to the
146 GUI until the user clicks the OK button to closes the dialog box).
147 *******************************************************************************/
148 void DialogBoxUtils::openWarningMessageBox( QString title, QString message )
149 {
150  QMessageBox messageBox;
151  messageBox.setWindowTitle( title );
152  messageBox.setText( message );
153  messageBox.setIcon( QMessageBox::Warning );
154  messageBox.exec();
155 }
