142 QString Help::getOpenGLInfo()
143 {
144  // Make OpenGL context current.
145  sourceGLWidget_->makeCurrent();
146  QString version( reinterpret_cast<const char *>(glGetString( GL_VERSION )) );
147  QString vendor( reinterpret_cast<const char *>(glGetString( GL_VENDOR )) );
148  QString renderer( reinterpret_cast<const char *>(glGetString( GL_RENDERER )) );
149  QString info( "OpenGL: Version " + version + "\n"
150  + "Graphics Card: " + renderer + "\n"
151  + "GPU Vendor: " + vendor + "\n\n"
152  + "Community Core Vision 1.3 requires OpenGL 2.0 or higher." );
153 
154  if( !isOpenGLv2Supported( version ) ) {
155  info += "\n\nYour machine appears to have an earlier version of OpenGL\n"
156  "and will probably not be able to run BigBlobby/CCV correctly.";
157  }
158  return info;
159 }

165 bool Help::isOpenGLv2Supported( const QString & version )
166 {
167  QString v = version.trimmed();
168 
169  if( v.size() > 0 ) {
170  v = v.left( 1 );
171  int n = v.toInt();
172  return (n >= 2);
173  }
174  return false;
175 }
