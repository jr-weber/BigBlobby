103 void FilterGLWidget::resizeGL( int viewportWidth, int viewportHeight )
104 {
105  int videoWidth = coreVision_->getVideoWidth(),
106  videoHeight = coreVision_->getVideoHeight();
107  float videoAspect = (float)videoWidth / videoHeight,
108  viewportAspect = (float)viewportWidth / viewportHeight;
109 
110  //debugPrintResizeGL( viewportWidth, viewportHeight, videoWidth, videoHeight );
111  glViewport( 0, 0, viewportWidth, viewportHeight );
112  glMatrixMode( GL_PROJECTION );
113  glLoadIdentity();
114 
115  if( viewportAspect > videoAspect ) {
116  // Viewport is too wide, so need some left/right padding (on x-axis)
117  int biggerWidth = (int)((viewportAspect / videoAspect) * videoWidth),
118  adjustment = (biggerWidth - videoWidth) / 2;
119  glOrtho( -adjustment, videoWidth + adjustment, 0, videoHeight, -1.0, 1.0 );
120  }
121  else if( videoAspect > viewportAspect ) {
122  // Viewport is too tall, so need some top/bottom padding (on y-axis).
123  int biggerHeight = (int)((videoAspect / viewportAspect) * videoHeight),
124  adjustment = (biggerHeight - videoHeight) / 2;
125  glOrtho( 0, videoWidth, -adjustment, videoHeight + adjustment, -1.0, 1.0 );
126  }
127  else { // Aspect ratios are equal.
128  glOrtho( 0, videoWidth, 0, videoHeight, -1.0, 1.0 );
129  }
130  glMatrixMode( GL_MODELVIEW );
131 }

153 /*******************************************************************************
154 This helper function for the constructor obtains the background color of the
155 window and stores it for use by the OpenGL rendering code.
156 
157 If the user resizes this widget to be something other than the 4:3 aspect ratio
158 for a VGA image, the image will still be rendered with a 4:3 aspect, and any
159 left over area will be filled with the background color. This approach was
160 used because Qt does not have any built-in way of restricting the aspect ratio
161 of a QGLWidget as it is resized.
162 *******************************************************************************/
163 void FilterGLWidget::initBackgroundColor()
164 {
165  QPalette pal = palette();
166  backgroundColor_ = pal.color( QPalette::Window );
167  debugPrintBackgroundColor();
168 }
